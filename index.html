# Load quarter ledger and compute revenue by segment

by_segment = ledger.groupby('segment')['revenue'].sum().reset_index()

# Sanity check: reported revenue must match system sum
reported = 2410000000  # from Finance
assert abs(by_segment['revenue'].sum() - reported) < 1e-6
